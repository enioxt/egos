import re
from datetime import datetime

def detect_emotion_trigger(text: str):
    keywords = {
        'raiva': ['irritado', 'ódio', 'raiva'],
        'tristeza': ['triste', 'desanimado', 'deprimido'],
        'felicidade': ['feliz', 'grato', 'alegre'],
        'ansiedade': ['ansioso', 'preocupado', 'inquieto'],
    }
    for emotion, terms in keywords.items():
        for term in terms:
            if re.search(rf'\b{term}\b', text.lower()):
                return {
                    'emotion': emotion,
                    'term': term,
                    'timestamp': datetime.now().isoformat()
                }
    return None

import yaml
import datetime

def carregar_memoria():
    with open("memory.yaml", "r") as f:
        return yaml.safe_load(f)

def limpar_memoria_sensivel():
    mem = carregar_memoria()
    hoje = datetime.datetime.now()

    for categoria, registros in mem.items():
        for chave, entradas in registros.items():
            mem[categoria][chave] = [
                r for r in entradas if (hoje - datetime.datetime.fromisoformat(r["timestamp"])).days < 30
            ]

    with open("memory.yaml", "w") as f:
        yaml.dump(mem, f)
