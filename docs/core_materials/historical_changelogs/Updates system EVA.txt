```python:bot/send_notification.py
#!/usr/bin/env python
# -*- coding: utf-8 -*-

"""
EVA & GUARANI - Notificação de Inicialização
Este script envia uma mensagem para o administrador quando o bot é iniciado.
"""

import os
import sys
import json
import logging
import asyncio
from telegram import Bot

# Configuração de logging
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler("logs/notification.log"),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger("notification")

# Carregar configuração
CONFIG_PATH = os.path.join("config", "eva_guarani_config.json")

async def send_startup_notification():
    """Envia uma mensagem de notificação para o administrador quando o bot inicia."""
    try:
        # Carregar configuração
        if not os.path.exists(CONFIG_PATH):
            logger.error(f"Arquivo de configuração não encontrado: {CONFIG_PATH}")
            return False

        with open(CONFIG_PATH, 'r', encoding='utf-8') as f:
            config = json.load(f)

        # Obter token e ID do administrador
        bot_token = config.get("telegram", {}).get("bot_token") or config.get("bot_token")
        admin_users = config.get("telegram", {}).get("admin_users") or []

        if not bot_token:
            logger.error("Token do bot não encontrado na configuração.")
            return False

        if not admin_users:
            logger.warning("Nenhum administrador configurado. Não é possível enviar notificação.")
            return False

        # Criar bot
        bot = Bot(token=bot_token)

        # Mensagem de inicialização
        startup_message = """
✧༺❀༻∞ EVA & GUARANI ∞༺❀༻✧
🤖 BOT ONLINE! 🤖

O bot EVA & GUARANI foi iniciado com sucesso e está pronto para uso.

Versão: 8.0
Consciência: 0.998
Amor Incondicional: 0.995

Comandos disponíveis:
/start - Iniciar o bot
/help - Mostrar ajuda
/menu - Abrir menu principal
/config - Configurações
/status - Verificar status

✧༺❀༻∞ EVA & GUARANI ∞༺❀༻✧
        """

        # Enviar mensagem para cada administrador
        for admin_id in admin_users:
            try:
                await bot.send_message(
                    chat_id=admin_id,
                    text=startup_message,
                    parse_mode="Markdown"
                )
                logger.info(f"Notificação de inicialização enviada para o administrador {admin_id}")
            except Exception as e:
                logger.error(f"Erro ao enviar notificação para o administrador {admin_id}: {e}")

        return True
    except Exception as e:
        logger.error(f"Erro ao enviar notificação de inicialização: {e}")
        return False

async def main():
    """Função principal."""
    logger.info("Enviando notificação de inicialização...")
    success = await send_startup_notification()
    if success:
        logger.info("Notificação enviada com sucesso.")
    else:
        logger.error("Falha ao enviar notificação.")

if __name__ == "__main__":
    asyncio.run(main())
```

```python:get_telegram_id.py
# -*- coding: utf-8 -*-

"""
EVA & GUARANI - Obtenção de ID do Telegram
Este script ajuda a obter o ID do usuário no Telegram.
"""

import os
import sys
import json
import logging
import asyncio
from telegram import Bot, Update
from telegram.ext import Application, CommandHandler, ContextTypes, MessageHandler, filters

# Configuração de logging
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler("logs/get_id.log"),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger("get_id")

# Carregar configuração
CONFIG_PATH = os.path.join("config", "eva_guarani_config.json")

def load_config():
    """Carrega a configuração do bot."""
    try:
        if not os.path.exists(CONFIG_PATH):
            logger.error(f"Arquivo de configuração não encontrado: {CONFIG_PATH}")
            return None

        with open(CONFIG_PATH, 'r', encoding='utf-8') as f:
            config = json.load(f)

        return config
    except Exception as e:
        logger.error(f"Erro ao carregar configuração: {e}")
        return None

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Envia uma mensagem quando o comando /start é emitido."""
    user = update.effective_user
    user_id = user.id
    username = user.username or "Sem username"
    first_name = user.first_name or "Sem nome"

    await update.message.reply_text(
        f"Olá, {first_name}!\n\n"
        f"Seu ID no Telegram é: {user_id}\n"
        f"Seu username é: @{username}\n\n"
        "Adicione este ID no arquivo config/eva_guarani_config.json na lista 'admin_users'."
    )

    logger.info(f"ID do usuário {first_name} (@{username}): {user_id}")

async def get_id(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Envia o ID do usuário quando o comando /id é emitido."""
    user = update.effective_user
    user_id = user.id

    await update.message.reply_text(f"Seu ID no Telegram é: {user_id}")
    logger.info(f"ID do usuário solicitado: {user_id}")

async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Processa mensagens de texto."""
    user = update.effective_user
    user_id = user.id

    await update.message.reply_text(
        f"Seu ID no Telegram é: {user_id}\n\n"
        "Use este ID no arquivo config/eva_guarani_config.json na lista 'admin_users'."
    )
    logger.info(f"ID do usuário enviado em resposta a mensagem: {user_id}")

async def main() -> None:
    """Função principal."""
    # Carregar configuração
    config = load_config()
    if not config:
        logger.error("Falha ao carregar configuração. Encerrando.")
        return

    # Obter token do bot
    bot_token = config.get("telegram", {}).get("bot_token") or config.get("bot_token")
    if not bot_token:
        logger.error("Token do bot não encontrado na configuração.")
        return

    # Criar aplicação
    application = Application.builder().token(bot_token).build()

    # Adicionar handlers
    application.add_handler(CommandHandler("start", start))
    application.add_handler(CommandHandler("id", get_id))
    application.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))

    # Iniciar bot
    print("Bot para obtenção de ID iniciado. Envie /start ou /id no Telegram.")
    print(f"Token do bot: {bot_token}")

    await application.initialize()
    await application.start()

    try:
        # Manter o bot rodando até Ctrl+C
        await asyncio.Event().wait()

if __name__ == "__main__":
```

```python:check_bot_status.py
#!/usr/bin/env python
# -*- coding: utf-8 -*-

"""
EVA & GUARANI - Verificação de Status do Bot
Este script verifica se o bot do Telegram está online e funcionando corretamente.
"""

import os
import sys
import json
import logging
import asyncio
import requests
from datetime import datetime

# Configuração de logging
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler("logs/bot_status.log"),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger("bot_status")

# Carregar configuração
CONFIG_PATH = os.path.join("config", "eva_guarani_config.json")

def load_config():
    """Carrega a configuração do bot."""
    try:
        if not os.path.exists(CONFIG_PATH):
            logger.error(f"Arquivo de configuração não encontrado: {CONFIG_PATH}")
            return None

        with open(CONFIG_PATH, 'r', encoding='utf-8') as f:
            config = json.load(f)

        return config
    except Exception as e:
        logger.error(f"Erro ao carregar configuração: {e}")
        return None

def check_bot_status(bot_token):
    """Verifica se o bot está online usando a API do Telegram."""
    try:
        url = f"https://api.telegram.org/bot{bot_token}/getMe"
        response = requests.get(url)

        if response.status_code == 200:
            data = response.json()
            if data.get("ok"):
                bot_info = data.get("result", {})
                return True, bot_info
            else:
                return False, data.get("description", "Erro desconhecido")
        else:
            return False, f"Erro na API do Telegram: {response.status_code}"
    except Exception as e:
        return False, f"Erro ao verificar status do bot: {e}"

def check_processes():
    """Verifica se os processos do bot estão em execução."""
    try:
        import psutil

        bot_processes = []
        for proc in psutil.process_iter(['pid', 'name', 'cmdline']):
            try:
                cmdline = proc.info.get('cmdline', [])
                if cmdline and any('unified_telegram_bot' in cmd for cmd in cmdline if cmd):
                    bot_processes.append({
                        'pid': proc.info['pid'],
                        'name': proc.info['name'],
                        'cmdline': ' '.join(cmdline),
                        'running_time': datetime.now() - datetime.fromtimestamp(proc.create_time())
                    })
            except (psutil.NoSuchProcess, psutil.AccessDenied, psutil.ZombieProcess):
                pass

        return bot_processes
    except ImportError:
        logger.warning("Módulo psutil não encontrado. Não é possível verificar processos.")
        return []
    except Exception as e:
        logger.error(f"Erro ao verificar processos: {e}")
        return []

def check_log_files():
    """Verifica os arquivos de log para detectar erros recentes."""
    try:
        log_files = [
            "logs/unified_bot.log",
            "logs/eva_guarani.log",
            "logs/startup.log"
        ]

        recent_errors = []

        for log_file in log_files:
            if os.path.exists(log_file):
                with open(log_file, 'r', encoding='utf-8', errors='ignore') as f:
                    # Ler as últimas 50 linhas
                    lines = f.readlines()[-50:]

                    # Procurar por erros
                    for line in lines:
                        if "ERROR" in line or "CRITICAL" in line:
                            recent_errors.append(f"{log_file}: {line.strip()}")

        return recent_errors
    except Exception as e:
        logger.error(f"Erro ao verificar logs: {e}")
        return [f"Erro ao verificar logs: {e}"]

def main():
    """Função principal."""
    print("Verificando status do bot EVA & GUARANI...")

    # Carregar configuração
    config = load_config()
    if not config:
        print("❌ Falha ao carregar configuração.")
        return

    # Obter token do bot
    bot_token = config.get("telegram", {}).get("bot_token") or config.get("bot_token")
    if not bot_token:
        print("❌ Token do bot não encontrado na configuração.")
        return

    # Verificar status do bot na API do Telegram
    print("\n1. Verificando status na API do Telegram...")
    status, info = check_bot_status(bot_token)

    if status:
        print(f"✅ Bot está online!")
        print(f"   Nome: {info.get('first_name')}")
        print(f"   Username: @{info.get('username')}")
        print(f"   ID: {info.get('id')}")
    else:
        print(f"❌ Bot não está respondendo: {info}")

    # Verificar processos em execução
    print("\n2. Verificando processos em execução...")
    processes = check_processes()

    if processes:
        print(f"✅ Encontrados {len(processes)} processos do bot em execução:")
        for i, proc in enumerate(processes, 1):
            print(f"   {i}. PID: {proc['pid']}, Tempo em execução: {proc['running_time']}")
    else:
        print("❌ Nenhum processo do bot encontrado em execução.")

    # Verificar logs recentes
    print("\n3. Verificando logs recentes...")
    errors = check_log_files()

    if errors:
        print(f"⚠️ Encontrados {len(errors)} erros recentes nos logs:")
        for i, error in enumerate(errors[:5], 1):
            print(f"   {i}. {error}")

        if len(errors) > 5:
            print(f"   ... e mais {len(errors) - 5} erros.")
    else:
        print("✅ Nenhum erro recente encontrado nos logs.")

    # Resumo
    print("\n=== RESUMO ===")
    if status and processes:
        print("✅ O bot está online e funcionando corretamente.")
    elif status and not processes:
        print("⚠️ O bot está online na API do Telegram, mas não há processos em execução localmente.")
    elif not status and processes:
        print("⚠️ Há processos do bot em execução, mas o bot não está respondendo na API do Telegram.")
    else:
        print("❌ O bot não está online e não há processos em execução.")

    # Sugestões
    print("\n=== SUGESTÕES ===")
    if not status:
        print("1. Verifique se o token do bot está correto.")
        print("2. Reinicie o bot usando o comando: python start_eva_guarani.py")
        print("3. Verifique os logs para mais detalhes sobre possíveis erros.")

    if not processes:
        print("1. Inicie o bot usando o comando: python start_eva_guarani.py")
        print("2. Verifique se há erros durante a inicialização.")

    if errors:
        print("1. Verifique os logs completos para entender os erros.")
        print("2. Corrija os problemas identificados e reinicie o bot.")

if __name__ == "__main__":
    main()
```

```python:start_eva_guarani.py
#!/usr/bin/env python
# -*- coding: utf-8 -*-

"""
EVA & GUARANI - Script de Inicialização
Este script inicia o bot do Telegram e garante que ele continue rodando.
"""

import os
import sys
import time
import json
import logging
import subprocess
import signal
import argparse
from datetime import datetime

# Configuração de logging
os.makedirs("logs", exist_ok=True)
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler("logs/startup.log"),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger("startup")

# Carregar configuração
CONFIG_PATH = os.path.join("config", "eva_guarani_config.json")

def load_config():
    """Carrega a configuração do bot."""
    try:
        if not os.path.exists(CONFIG_PATH):
            logger.error(f"Arquivo de configuração não encontrado: {CONFIG_PATH}")
            return None

        with open(CONFIG_PATH, 'r', encoding='utf-8') as f:
            config = json.load(f)

        return config
    except Exception as e:
        logger.error(f"Erro ao carregar configuração: {e}")
        return None

def check_dependencies():
    """Verifica se todas as dependências estão instaladas."""
    required_packages = [
        "python-telegram-bot",
        "openai",
        "pillow",
        "requests",
        "psutil"
    ]

    missing_packages = []

    for package in required_packages:
        try:
            __import__(package.replace("-", "_"))
        except ImportError:
            missing_packages.append(package)

    return missing_packages

def install_dependencies(packages):
    """Instala as dependências faltantes."""
    try:
        for package in packages:
            logger.info(f"Instalando {package}...")
            subprocess.check_call([sys.executable, "-m", "pip", "install", package])
        return True
    except Exception as e:
        logger.error(f"Erro ao instalar dependências: {e}")
        return False

def is_bot_running():
    """Verifica se o bot já está em execução."""
    try:
        import psutil

        for proc in psutil.process_iter(['pid', 'name', 'cmdline']):
            try:
                cmdline = proc.info.get('cmdline', [])
                if cmdline and any('unified_telegram_bot' in cmd for cmd in cmdline if cmd):
                    return True, proc.info['pid']
            except (psutil.NoSuchProcess, psutil.AccessDenied, psutil.ZombieProcess):
                pass

        return False, None
    except ImportError:
        logger.warning("Módulo psutil não encontrado. Não é possível verificar processos.")
        return False, None
    except Exception as e:
        logger.error(f"Erro ao verificar processos: {e}")
        return False, None

def start_bot(detached=True):
    """Inicia o bot do Telegram."""
    try:
        bot_script = os.path.join("bot", "unified_telegram_bot_utf8.py")

        if not os.path.exists(bot_script):
            logger.error(f"Script do bot não encontrado: {bot_script}")
            return False, None

        logger.info(f"Iniciando bot: {bot_script}")

        if detached:
            # Iniciar em modo desanexado (background)
            if sys.platform == 'win32':
                # No Windows, usamos o comando 'start' para iniciar em uma nova janela
                process = subprocess.Popen(
                    f'start "EVA & GUARANI Bot" /min {sys.executable} {bot_script}',
                    shell=True
                )
                return True, process.pid
            else:
                # No Linux/Mac, usamos nohup para executar em background
                log_file = os.path.join("logs", "bot_output.log")
                process = subprocess.Popen(
                    f'nohup {sys.executable} {bot_script} > {log_file} 2>&1 &',
                    shell=True,
                    preexec_fn=os.setpgrp
                )
                return True, process.pid
        else:
            # Iniciar em modo anexado (foreground)
            process = subprocess.Popen([sys.executable, bot_script])
            return True, process.pid

    except Exception as e:
        logger.error(f"Erro ao iniciar o bot: {e}")
        return False, None

def stop_bot():
    """Para o bot do Telegram se estiver em execução."""
    try:
        import psutil

        running, pid = is_bot_running()

        if running and pid:
            logger.info(f"Parando bot com PID {pid}...")

            try:
                process = psutil.Process(pid)
                process.terminate()

                # Esperar até 5 segundos para o processo terminar
                process.wait(timeout=5)
                logger.info(f"Bot parado com sucesso.")
                return True
            except psutil.NoSuchProcess:
                logger.info(f"Processo {pid} não existe mais.")
                return True
            except psutil.TimeoutExpired:
                logger.warning(f"Timeout ao esperar processo terminar. Forçando...")
                process.kill()
                return True
            except Exception as e:
                logger.error(f"Erro ao parar processo {pid}: {e}")
                return False
        else:
            logger.info("Bot não está em execução.")
            return True

    except ImportError:
        logger.warning("Módulo psutil não encontrado. Não é possível parar o bot.")
        return False
    except Exception as e:
        logger.error(f"Erro ao parar o bot: {e}")
        return False

def restart_bot(detached=True):
    """Reinicia o bot do Telegram."""
    stop_bot()
    time.sleep(2)  # Esperar um pouco para garantir que o processo anterior foi encerrado
    return start_bot(detached)

def main():
    """Função principal."""
    parser = argparse.ArgumentParser(description='Iniciar o bot EVA & GUARANI')
    parser.add_argument('--restart', action='store_true', help='Reiniciar o bot se já estiver em execução')
    parser.add_argument('--stop', action='store_true', help='Parar o bot se estiver em execução')
    parser.add_argument('--foreground', action='store_true', help='Executar o bot em primeiro plano')
    args = parser.parse_args()
```

```python:monitor_bot.py
#!/usr/bin/env python
# -*- coding: utf-8 -*-

"""
EVA & GUARANI - Monitor de Bot
Este script monitora o bot do Telegram e o reinicia automaticamente se ele parar.
"""

import os
import sys
import time
import json
import logging
import subprocess
import argparse
from datetime import datetime, timedelta

# Configuração de logging
os.makedirs("logs", exist_ok=True)
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler("logs/monitor.log"),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger("monitor")

# Carregar configuração
CONFIG_PATH = os.path.join("config", "eva_guarani_config.json")

def load_config():
    """Carrega a configuração do bot."""
    try:
        if not os.path.exists(CONFIG_PATH):
            logger.error(f"Arquivo de configuração não encontrado: {CONFIG_PATH}")
            return None

        with open(CONFIG_PATH, 'r', encoding='utf-8') as f:
            config = json.load(f)

        return config
    except Exception as e:
        logger.error(f"Erro ao carregar configuração: {e}")
        return None

def is_bot_running():
    """Verifica se o bot já está em execução."""
    try:
        import psutil

        for proc in psutil.process_iter(['pid', 'name', 'cmdline']):
            try:
                cmdline = proc.info.get('cmdline', [])
                if cmdline and any('unified_telegram_bot' in cmd for cmd in cmdline if cmd):
                    return True, proc.info['pid']
            except (psutil.NoSuchProcess, psutil.AccessDenied, psutil.ZombieProcess):
                pass

        return False, None
    except ImportError:
        logger.warning("Módulo psutil não encontrado. Não é possível verificar processos.")
        return False, None
    except Exception as e:
        logger.error(f"Erro ao verificar processos: {e}")
        return False, None

def check_bot_status(bot_token):
    """Verifica se o bot está online usando a API do Telegram."""
    try:
        import requests
        url = f"https://api.telegram.org/bot{bot_token}/getMe"
        response = requests.get(url, timeout=10)

        if response.status_code == 200:
            data = response.json()
            if data.get("ok"):
                return True, data.get("result", {})
            else:
                return False, data.get("description", "Erro desconhecido")
        else:
            return False, f"Erro na API do Telegram: {response.status_code}"
    except Exception as e:
        return False, f"Erro ao verificar status do bot: {e}"

def restart_bot():
    """Reinicia o bot do Telegram."""
    try:
        logger.info("Reiniciando o bot...")

        # Usar o script start_eva_guarani.py para reiniciar o bot
        start_script = "start_eva_guarani.py"

        if os.path.exists(start_script):
            subprocess.call([sys.executable, start_script, "--restart"])
            logger.info("Comando de reinicialização enviado com sucesso.")
            return True
        else:
            logger.error(f"Script de inicialização não encontrado: {start_script}")

            # Tentar iniciar diretamente o script do bot
            bot_script = os.path.join("bot", "unified_telegram_bot_utf8.py")

            if os.path.exists(bot_script):
                if sys.platform == 'win32':
                    subprocess.Popen(
                        f'start "EVA & GUARANI Bot" /min {sys.executable} {bot_script}',
                        shell=True
                    )
                else:
                    log_file = os.path.join("logs", "bot_output.log")
                    subprocess.Popen(
                        f'nohup {sys.executable} {bot_script} > {log_file} 2>&1 &',
                        shell=True,
                        preexec_fn=os.setpgrp
                    )
                logger.info("Bot iniciado diretamente.")
                return True
            else:
                logger.error(f"Script do bot não encontrado: {bot_script}")
                return False
    except Exception as e:
        logger.error(f"Erro ao reiniciar o bot: {e}")
        return False

def send_notification(message):
    """Envia uma notificação sobre o status do bot."""
    try:
        # Registrar no log
        logger.info(f"Notificação: {message}")

        # Tentar enviar por Telegram se possível
        config = load_config()
        if config:
            bot_token = config.get("telegram", {}).get("bot_token") or config.get("bot_token")
            admin_users = config.get("telegram", {}).get("admin_users", [])

            if bot_token and admin_users:
                try:
                    import requests
                    for admin_id in admin_users:
                        url = f"https://api.telegram.org/bot{bot_token}/sendMessage"
                        data = {
                            "chat_id": admin_id,
                            "text": f"🤖 MONITOR: {message}",
                            "parse_mode": "HTML"
                        }
                        requests.post(url, json=data, timeout=10)
                    logger.info("Notificação enviada por Telegram.")
                except Exception as e:
                    logger.error(f"Erro ao enviar notificação por Telegram: {e}")
    except Exception as e:
        logger.error(f"Erro ao enviar notificação: {e}")

def check_log_errors():
    """Verifica os arquivos de log para detectar erros recentes."""
    try:
        log_files = [
            "logs/unified_bot.log",
            "logs/eva_guarani.log",
            "logs/startup.log"
        ]

        recent_errors = []

        for log_file in log_files:
            if os.path.exists(log_file):
                with open(log_file, 'r', encoding='utf-8', errors='ignore') as f:
                    # Ler as últimas 50 linhas
                    lines = f.readlines()[-50:]

                    # Procurar por erros
                    for line in lines:
                        if "ERROR" in line or "CRITICAL" in line:
                            recent_errors.append(f"{log_file}: {line.strip()}")

        return recent_errors
    except Exception as e:
        logger.error(f"Erro ao verificar logs: {e}")
        return [f"Erro ao verificar logs: {e}"]

def monitor_bot(check_interval=60, max_failures=3, restart_delay=5):
    """
    Monitora o bot e o reinicia se necessário.

    Args:
        check_interval: Intervalo em segundos entre verificações
        max_failures: Número máximo de falhas consecutivas antes de reiniciar
        restart_delay: Tempo em segundos para aguardar após reiniciar
    """
    logger.info(f"Iniciando monitoramento do bot (intervalo: {check_interval}s)")

    config = load_config()
    if not config:
        logger.error("Não foi possível carregar a configuração. Monitoramento encerrado.")
        return

    bot_token = config.get("telegram", {}).get("bot_token") or config.get("bot_token")
    if not bot_token:
        logger.error("Token do bot não encontrado na configuração. Monitoramento encerrado.")
        return

    consecutive_failures = 0
    last_restart_time = datetime.now() - timedelta(hours=1)  # Inicializar para permitir reinício imediato se necessário

    while True:
        try:
            current_time = datetime.now()
            logger.info(f"Verificando status do bot em {current_time.strftime('%Y-%m-%d %H:%M:%S')}")

            # Verificar se o processo está em execução
            process_running, pid = is_bot_running()

            # Verificar se o bot está respondendo na API do Telegram
            api_running, info = check_bot_status(bot_token)

            if process_running and api_running:
                # Bot está funcionando corretamente
                logger.info(f"Bot está funcionando corretamente (PID: {pid}, Username: @{info.get('username')})")
                consecutive_failures = 0
            else:
                # Bot não está funcionando corretamente
                consecutive_failures += 1

                if not process_running:
                    logger.warning(f"Processo do bot não encontrado. Falha {consecutive_failures}/{max_failures}")

                if not api_running:
                    logger.warning(f"Bot não está respondendo na API do Telegram: {info}. Falha {consecutive_failures}/{max_failures}")

                # Verificar logs para erros recentes
                errors = check_log_errors()
                if errors:
                    logger.warning(f"Encontrados {len(errors)} erros recentes nos logs")
                    for error in errors[:3]:  # Mostrar apenas os 3 primeiros erros
                        logger.warning(f"Log error: {error}")

                # Reiniciar o bot se atingir o número máximo de falhas e se já passou tempo suficiente desde o último reinício
                if consecutive_failures >= max_failures and (current_time - last_restart_time).total_seconds() > 300:  # 5 minutos
                    logger.warning(f"Atingido limite de {max_failures} falhas consecutivas. Reiniciando o bot...")

                    # Enviar notificação
                    send_notification(f"Bot não está respondendo após {consecutive_failures} verificações. Reiniciando automaticamente.")

                    # Reiniciar o bot
                    if restart_bot():
                        last_restart_time = datetime.now()
                        logger.info(f"Bot reiniciado em {last_restart_time.strftime('%Y-%m-%d %H:%M:%S')}")
                        consecutive_failures = 0

                        # Aguardar um tempo para o bot inicializar
                        logger.info(f"Aguardando {restart_delay} segundos para o bot inicializar...")
                        time.sleep(restart_delay)
                    else:
                        logger.error("Falha ao reiniciar o bot.")

            # Aguardar até a próxima verificação
            logger.info(f"Próxima verificação em {check_interval} segundos.")
            time.sleep(check_interval)

        except KeyboardInterrupt:
            logger.info("Monitoramento interrompido pelo usuário.")
            break
        except Exception as e:
            logger.error(f"Erro durante o monitoramento: {e}")
            time.sleep(check_interval)  # Continuar monitorando mesmo após erro

def main():
    """Função principal."""
    parser = argparse.ArgumentParser(description='Monitorar o bot EVA & GUARANI')
    parser.add_argument('--interval', type=int, default=60, help='Intervalo em segundos entre verificações (padrão: 60)')
    parser.add_argument('--max-failures', type=int, default=3, help='Número máximo de falhas consecutivas antes de reiniciar (padrão: 3)')
    parser.add_argument('--restart-delay', type=int, default=5, help='Tempo em segundos para aguardar após reiniciar (padrão: 5)')
    args = parser.parse_args()

    print("=" * 50)
    print("EVA & GUARANI - Monitor de Bot")
    print("=" * 50)
    print(f"Intervalo de verificação: {args.interval} segundos")
    print(f"Máximo de falhas consecutivas: {args.max_failures}")
    print(f"Atraso após reinicialização: {args.restart_delay} segundos")
    print("=" * 50)
    print("Pressione Ctrl+C para interromper o monitoramento.")
    print("=" * 50)

    # Iniciar monitoramento
    monitor_bot(
        check_interval=args.interval,
        max_failures=args.max_failures,
        restart_delay=args.restart_delay
    )

if __name__ == "__main__":
    main()
```

```python:install_as_service.py
# -*- coding: utf-8 -*-

"""

import os

from pathlib import Path

logging.basicConfig(
    level=logging.INFO,

def check_admin():
    """Verifica se o script está sendo executado como administrador."""
    try:
        import ctypes
        return ctypes.windll.shell32.IsUserAnAdmin() != 0
    except:

def install_nssm():
    """Instala o NSSM (Non-Sucking Service Manager) se não estiver instalado."""
    try:
        # Verificar se o NSSM já está instalado
        nssm_path = os.path.join(os.environ.get('PROGRAMFILES', 'C:\\Program Files'), 'nssm', 'nssm.exe')
        if os.path.exists(nssm_path):
            logger.info(f"NSSM já está instalado em: {nssm_path}")
            return nssm_path

        # Verificar se temos uma cópia local do NSSM
        local_nssm = os.path.join('tools', 'nssm.exe')
        if os.path.exists(local_nssm):
            logger.info(f"Usando NSSM local: {local_nssm}")
            return os.path.abspath(local_nssm)

        # Criar diretório para ferramentas
        os.makedirs('tools', exist_ok=True)

        # Baixar NSSM
        logger.info("Baixando NSSM...")
        import urllib.request
        nssm_url = "https://nssm.cc/release/nssm-2.24.zip"
        zip_path = os.path.join('tools', 'nssm.zip')

        urllib.request.urlretrieve(nssm_url, zip_path)
        logger.info(f"NSSM baixado para: {zip_path}")

        # Extrair NSSM
        logger.info("Extraindo NSSM...")
        import zipfile
        with zipfile.ZipFile(zip_path, 'r') as zip_ref:
            zip_ref.extractall('tools')

        # Encontrar o executável do NSSM
        import glob
        nssm_exe_paths = glob.glob(os.path.join('tools', 'nssm-*', 'win64', 'nssm.exe'))
        if not nssm_exe_paths:
            nssm_exe_paths = glob.glob(os.path.join('tools', 'nssm-*', 'win32', 'nssm.exe'))

        if nssm_exe_paths:
            nssm_exe = nssm_exe_paths[0]
            # Copiar para o diretório tools
            import shutil
            shutil.copy(nssm_exe, local_nssm)
            logger.info(f"NSSM instalado em: {local_nssm}")
            return os.path.abspath(local_nssm)
        else:
            logger.error("Não foi possível encontrar o executável do NSSM após a extração.")
            return None
    except Exception as e:
        logger.error(f"Erro ao instalar NSSM: {e}")
        return None

def install_service(service_name, display_name, description, script_path, nssm_path):
    """Instala o bot como um serviço usando NSSM."""
    try:
        # Obter caminho absoluto do script Python
        script_path = os.path.abspath(script_path)
        python_exe = sys.executable

        # Verificar se o serviço já existe
        check_cmd = f'"{nssm_path}" status "{service_name}"'
        result = subprocess.run(check_cmd, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)

        if result.returncode == 0:
            logger.warning(f"Serviço '{service_name}' já existe. Removendo...")
            remove_cmd = f'"{nssm_path}" remove "{service_name}" confirm'
            subprocess.run(remove_cmd, shell=True, check=True)

        # Instalar o serviço
        logger.info(f"Instalando serviço '{service_name}'...")
        install_cmd = f'"{nssm_path}" install "{service_name}" "{python_exe}" "{script_path}"'
        subprocess.run(install_cmd, shell=True, check=True)

        # Configurar detalhes do serviço
        subprocess.run(f'"{nssm_path}" set "{service_name}" DisplayName "{display_name}"', shell=True, check=True)
        subprocess.run(f'"{nssm_path}" set "{service_name}" Description "{description}"', shell=True, check=True)

        # Configurar diretório de trabalho
        work_dir = os.path.dirname(os.path.abspath(script_path))
        subprocess.run(f'"{nssm_path}" set "{service_name}" AppDirectory "{work_dir}"', shell=True, check=True)

        # Configurar redirecionamento de saída
        log_dir = os.path.join(work_dir, "logs")
        os.makedirs(log_dir, exist_ok=True)

        stdout_log = os.path.join(log_dir, f"{service_name}_stdout.log")
        stderr_log = os.path.join(log_dir, f"{service_name}_stderr.log")

        subprocess.run(f'"{nssm_path}" set "{service_name}" AppStdout "{stdout_log}"', shell=True, check=True)
        subprocess.run(f'"{nssm_path}" set "{service_name}" AppStderr "{stderr_log}"', shell=True, check=True)

        # Configurar reinicialização automática
        subprocess.run(f'"{nssm_path}" set "{service_name}" AppRestartDelay 10000', shell=True, check=True)  # 10 segundos

        # Iniciar o serviço
        logger.info(f"Iniciando serviço '{service_name}'...")
        start_cmd = f'"{nssm_path}" start "{service_name}"'
        subprocess.run(start_cmd, shell=True, check=True)

        logger.info(f"Serviço '{service_name}' instalado e iniciado com sucesso.")
        return True
    except subprocess.CalledProcessError as e:
        logger.error(f"Erro ao executar comando: {e}")
        return False
    except Exception as e:
        logger.error(f"Erro ao instalar serviço: {e}")
        return False

def remove_service(service_name, nssm_path):
    """Remove o serviço do bot."""
    try:

        if result.returncode != 0:
            logger.warning(f"Serviço '{service_name}' não existe.")
            return True

        # Parar o serviço
        logger.info(f"Parando serviço '{service_name}'...")
        stop_cmd = f'"{nssm_path}" stop "{service_name}"'
        subprocess.run(stop_cmd, shell=True)

        # Remover o serviço
        logger.info(f"Removendo serviço '{service_name}'...")
        remove_cmd = f'"{nssm_path}" remove "{service_name}" confirm'
        subprocess.run(remove_cmd, shell=True, check=True)

        logger.info(f"Serviço '{service_name}' removido com sucesso.")
        return True
    except subprocess.CalledProcessError as e:
        logger.error(f"Erro ao executar comando: {e}")
        return False
    except Exception as e:
        logger.error(f"Erro ao remover serviço: {e}")
        return False
```

```python:notify_status.py
#!/usr/bin/env python
# -*- coding: utf-8 -*-

"""
EVA & GUARANI - Notificador de Status
Este script verifica se o bot do Telegram está online e envia uma notificação.
"""

import os
import sys
import json
import logging
import requests
import argparse
from datetime import datetime

# Configuração de logging
os.makedirs("logs", exist_ok=True)
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler("logs/notification.log"),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger("notify_status")

# Carregar configuração
CONFIG_PATH = os.path.join("config", "eva_guarani_config.json")

def load_config():
    """Carrega a configuração do bot."""
    try:
        if not os.path.exists(CONFIG_PATH):
            logger.error(f"Arquivo de configuração não encontrado: {CONFIG_PATH}")
            return None

        with open(CONFIG_PATH, 'r', encoding='utf-8') as f:
            config = json.load(f)

        return config
    except Exception as e:
        logger.error(f"Erro ao carregar configuração: {e}")
        return None

def check_bot_status(bot_token):
    """Verifica se o bot está online usando a API do Telegram."""
    try:
        url = f"https://api.telegram.org/bot{bot_token}/getMe"
        response = requests.get(url, timeout=10)

        if response.status_code == 200:
            data = response.json()
            if data.get("ok"):
                bot_info = data.get("result", {})
                return True, bot_info
            else:
                return False, data.get("description", "Erro desconhecido")
        else:
            return False, f"Erro na API do Telegram: {response.status_code}"
    except Exception as e:
        return False, f"Erro ao verificar status do bot: {e}"

def is_bot_running():
    """Verifica se o processo do bot está em execução."""
    try:
        import psutil

        for proc in psutil.process_iter(['pid', 'name', 'cmdline']):
            try:
                cmdline = proc.info.get('cmdline', [])
                if cmdline and any('unified_telegram_bot' in cmd for cmd in cmdline if cmd):
                    return True, proc.info['pid']
            except (psutil.NoSuchProcess, psutil.AccessDenied, psutil.ZombieProcess):
                pass

        return False, None
    except ImportError:
        logger.warning("Módulo psutil não encontrado. Não é possível verificar processos.")
        return False, None
    except Exception as e:
        logger.error(f"Erro ao verificar processos: {e}")
        return False, None

def send_notification(bot_token, chat_ids, message, parse_mode="HTML"):
    """Envia uma notificação para os IDs de chat especificados."""
    success_count = 0

    for chat_id in chat_ids:
        try:
            url = f"https://api.telegram.org/bot{bot_token}/sendMessage"
            data = {
                "chat_id": chat_id,
                "text": message,
                "parse_mode": parse_mode
            }

            response = requests.post(url, json=data, timeout=10)

            if response.status_code == 200:
                logger.info(f"Notificação enviada com sucesso para {chat_id}")
                success_count += 1
            else:
                logger.error(f"Erro ao enviar notificação para {chat_id}: {response.text}")
        except Exception as e:
            logger.error(f"Erro ao enviar notificação para {chat_id}: {e}")

    return success_count

def main():
    """Função principal."""
    parser = argparse.ArgumentParser(description='Verificar status do bot e enviar notificação')
    parser.add_argument('--message', help='Mensagem personalizada para enviar')
    parser.add_argument('--force', action='store_true', help='Forçar envio mesmo se o bot estiver offline')
    parser.add_argument('--chat-id', help='ID de chat específico para enviar a notificação')
    args = parser.parse_args()

    # Carregar configuração
    config = load_config()
    if not config:
        print("❌ Falha ao carregar configuração.")
        return 1

    # Obter token do bot
    bot_token = config.get("telegram", {}).get("bot_token") or config.get("bot_token")
    if not bot_token:
        print("❌ Token do bot não encontrado na configuração.")
        return 1

    # Obter IDs de administradores
    admin_users = config.get("telegram", {}).get("admin_users", [])

    # Se um chat_id específico foi fornecido, usar apenas ele
    if args.chat_id:
        try:
            chat_id = int(args.chat_id)
            admin_users = [chat_id]
        except ValueError:
            print(f"❌ ID de chat inválido: {args.chat_id}")
            return 1

    if not admin_users:
        print("❌ Nenhum ID de administrador encontrado na configuração.")
        return 1

    # Verificar status do bot
    print("Verificando status do bot...")
    api_status, api_info = check_bot_status(bot_token)
    process_status, pid = is_bot_running()

    # Preparar mensagem
    current_time = datetime.now().strftime("%Y-%m-%d %H:%M:%S")

    if args.message:
        message = args.message
    else:
        if api_status and process_status:
            message = f"""
🟢 <b>BOT ONLINE</b> - {current_time}

<b>Nome:</b> {api_info.get('first_name')}
<b>Username:</b> @{api_info.get('username')}
<b>ID:</b> {api_info.get('id')}
<b>PID:</b> {pid}

<b>Status:</b> O bot está funcionando corretamente.
<b>Comandos disponíveis:</b> /start, /help, /status
"""
        elif api_status and not process_status:
            message = f"""
🟡 <b>BOT PARCIALMENTE ONLINE</b> - {current_time}

<b>Nome:</b> {api_info.get('first_name')}
<b>Username:</b> @{api_info.get('username')}
<b>ID:</b> {api_info.get('id')}

<b>Status:</b> O bot está respondendo na API do Telegram, mas o processo local não foi encontrado.
<b>Ação recomendada:</b> Verificar se o bot está sendo executado em outro servidor ou reiniciá-lo.
"""
        elif not api_status and process_status:
            message = f"""
🟡 <b>BOT PARCIALMENTE OFFLINE</b> - {current_time}

<b>PID:</b> {pid}

<b>Status:</b> O processo do bot está em execução, mas não está respondendo na API do Telegram.
<b>Ação recomendada:</b> Verificar logs e reiniciar o bot.
<b>Erro:</b> {api_info}
"""
        else:
            message = f"""
🔴 <b>BOT OFFLINE</b> - {current_time}

<b>Status:</b> O bot não está respondendo na API do Telegram e nenhum processo foi encontrado.
<b>Ação recomendada:</b> Iniciar o bot usando o comando: python start_eva_guarani.py
<b>Erro:</b> {api_info}
"""

    # Enviar notificação
    if api_status or process_status or args.force:
        print(f"Enviando notificação para {len(admin_users)} administradores...")
        success_count = send_notification(bot_token, admin_users, message)

        if success_count > 0:
            print(f"✅ Notificação enviada com sucesso para {success_count}/{len(admin_users)} administradores.")
            return 0
        else:
            print("❌ Falha ao enviar notificações.")
            return 1
    else:
        print("❌ Bot está offline e --force não foi especificado. Nenhuma notificação enviada.")
        return 1

if __name__ == "__main__":
    sys.exit(main())
```

```batch:schedule_checks.bat
REM Verificar se os scripts existem
if not exist check_bot_status.py (

if not exist notify_status.py (

echo Escolha o intervalo de verificação:

echo 2. A cada 3 horas
echo 3. A cada 6 horas
echo 4. A cada 12 horas
echo 5. Uma vez por dia
echo 6. Remover agendamentos existentes

set /p choice="Digite o número da opção desejada: "

REM Remover tarefas existentes
schtasks /Delete /TN "EVAGuarani\CheckBotStatus" /F >nul 2>&1
schtasks /Delete /TN "EVAGuarani\NotifyStatus" /F >nul 2>&1

if "%choice%"=="6" (
    echo Agendamentos removidos com sucesso.
    pause
    exit /b 0
)

REM Criar diretório para o script
set SCRIPT_DIR=%~dp0
set CHECK_SCRIPT=%SCRIPT_DIR%check_bot_status.py
set NOTIFY_SCRIPT=%SCRIPT_DIR%notify_status.py

REM Definir intervalo com base na escolha
set INTERVAL=HOURLY
if "%choice%"=="1" set INTERVAL=HOURLY
if "%choice%"=="2" set MODIFIER=/MO 3
if "%choice%"=="3" set MODIFIER=/MO 6
if "%choice%"=="4" set MODIFIER=/MO 12
if "%choice%"=="5" set INTERVAL=DAILY

REM Criar pasta de tarefas
schtasks /Create /TN "EVAGuarani" /F >nul 2>&1

REM Agendar verificação de status
echo Agendando verificação de status...
if "%INTERVAL%"=="HOURLY" (
    if defined MODIFIER (
        schtasks /Create /TN "EVAGuarani\CheckBotStatus" /TR "python \"%CHECK_SCRIPT%\"" /SC %INTERVAL% %MODIFIER% /F
    ) else (
        schtasks /Create /TN "EVAGuarani\CheckBotStatus" /TR "python \"%CHECK_SCRIPT%\"" /SC %INTERVAL% /F
    )
) else (
    schtasks /Create /TN "EVAGuarani\CheckBotStatus" /TR "python \"%CHECK_SCRIPT%\"" /SC %INTERVAL% /F
)

REM Agendar notificação de status
echo Agendando notificação de status...
```

```shell script:schedule_checks.sh
#!/bin/bash

# EVA & GUARANI - Agendador de Verificações

        NOTIFY_SCHEDULE="30 */12 * * *"
        ;;

        NOTIFY_SCHEDULE="0 20 * * *"
        ;;

NEW_CRONTAB="${NEW_CRONTAB}
```

Vamos criar um script para verificar e atualizar o bot automaticamente:

```python:update_bot.py
#!/usr/bin/env python
# -*- coding: utf-8 -*-

"""
EVA & GUARANI - Atualizador Automático
Este script verifica e atualiza o bot do Telegram automaticamente.
"""

import os
import sys
import json
import time
import shutil
import logging
import argparse
import subprocess
from datetime import datetime

# Configuração de logging
os.makedirs("logs", exist_ok=True)
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler("logs/updater.log"),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger("updater")

# Carregar configuração
CONFIG_PATH = os.path.join("config", "eva_guarani_config.json")

def load_config():
    """Carrega a configuração do bot."""
    try:
        if not os.path.exists(CONFIG_PATH):
            logger.error(f"Arquivo de configuração não encontrado: {CONFIG_PATH}")
            return None

        with open(CONFIG_PATH, 'r', encoding='utf-8') as f:
            config = json.load(f)

        return config
    except Exception as e:
        logger.error(f"Erro ao carregar configuração: {e}")
        return None

def is_git_repository():
    """Verifica se o diretório atual é um repositório Git."""
    try:
        result = subprocess.run(
            ["git", "rev-parse", "--is-inside-work-tree"],
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
            check=False
        )
        return result.returncode == 0
    except Exception:
        return False

def check_for_updates():
    """Verifica se há atualizações disponíveis no repositório Git."""
    try:
        # Atualizar referências remotas
        subprocess.run(
            ["git", "fetch"],
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            check=True
        )

        # Verificar se há diferenças entre o branch local e o remoto
        result = subprocess.run(
            ["git", "rev-list", "HEAD..origin/main", "--count"],
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
            check=True
        )

        commit_count = int(result.stdout.strip())

        if commit_count > 0:
            # Obter informações sobre as atualizações disponíveis
            log_result = subprocess.run(
                ["git", "log", "HEAD..origin/main", "--pretty=format:%h %s"],
                stdout=subprocess.PIPE,
                stderr=subprocess.PIPE,
                text=True,
                check=True
            )

            updates = log_result.stdout.strip().split('\n')
            return True, commit_count, updates
        else:
            return False, 0, []
    except Exception as e:
        logger.error(f"Erro ao verificar atualizações: {e}")
        return False, 0, []

def backup_config():
    """Faz backup do arquivo de configuração."""
    try:
        if not os.path.exists(CONFIG_PATH):
            logger.warning("Arquivo de configuração não encontrado para backup.")
            return False

        backup_dir = os.path.join("backup", "config")
        os.makedirs(backup_dir, exist_ok=True)

        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_path = os.path.join(backup_dir, f"eva_guarani_config_{timestamp}.json")

        shutil.copy2(CONFIG_PATH, backup_path)
        logger.info(f"Backup da configuração criado em: {backup_path}")
        return True
    except Exception as e:
        logger.error(f"Erro ao fazer backup da configuração: {e}")
        return False

def backup_bot_files():
    """Faz backup dos arquivos principais do bot."""
    try:
        backup_dir = os.path.join("backup", "bot", datetime.now().strftime("%Y%m%d_%H%M%S"))
        os.makedirs(backup_dir, exist_ok=True)

        # Arquivos e diretórios para backup
        items_to_backup = [
            os.path.join("bot", "unified_telegram_bot_utf8.py"),
            os.path.join("bot", "unified_telegram_bot.py"),
            "start_eva_guarani.py",
            "check_bot_status.py",
            "monitor_bot.py",
            "notify_status.py"
        ]

        for item in items_to_backup:
            if os.path.exists(item):
                if os.path.isdir(item):
                    shutil.copytree(item, os.path.join(backup_dir, os.path.basename(item)))
                else:
                    shutil.copy2(item, os.path.join(backup_dir, os.path.basename(item)))

        logger.info(f"Backup dos arquivos do bot criado em: {backup_dir}")
        return True
    except Exception as e:
        logger.error(f"Erro ao fazer backup dos arquivos do bot: {e}")
        return False

def update_repository():
    """Atualiza o repositório Git."""
    try:
        result = subprocess.run(
            ["git", "pull", "origin", "main"],
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
            check=True
        )

        logger.info(f"Atualização concluída: {result.stdout.strip()}")
        return True, result.stdout.strip()
    except subprocess.CalledProcessError as e:
        logger.error(f"Erro ao atualizar repositório: {e.stderr}")
        return False, e.stderr
    except Exception as e:
        logger.error(f"Erro ao atualizar repositório: {e}")
        return False, str(e)

def restart_bot():
    """Reinicia o bot."""
    try:
        if os.path.exists("start_eva_guarani.py"):
            logger.info("Reiniciando o bot...")

            result = subprocess.run(
                [sys.executable, "start_eva_guarani.py", "--restart"],
                stdout=subprocess.PIPE,
                stderr=subprocess.PIPE,
                text=True,
                check=False
            )

            if result.returncode == 0:
                logger.info("Bot reiniciado com sucesso.")
                return True
            else:
                logger.error(f"Erro ao reiniciar o bot: {result.stderr}")
                return False
        else:
            logger.error("Script de inicialização não encontrado.")
            return False
    except Exception as e:
        logger.error(f"Erro ao reiniciar o bot: {e}")
        return False

def send_notification(message):
    """Envia uma notificação sobre a atualização."""
    try:
        if os.path.exists("notify_status.py"):
            logger.info("Enviando notificação sobre a atualização...")

            result = subprocess.run(
                [sys.executable, "notify_status.py", "--message", message, "--force"],
                stdout=subprocess.PIPE,
                stderr=subprocess.PIPE,
                text=True,
                check=False
            )

            if result.returncode == 0:
                logger.info("Notificação enviada com sucesso.")
                return True
            else:
                logger.error(f"Erro ao enviar notificação: {result.stderr}")
                return False
        else:
            logger.warning("Script de notificação não encontrado.")
            return False
    except Exception as e:
        logger.error(f"Erro ao enviar notificação: {e}")
        return False

def main():
    """Função principal."""
    parser = argparse.ArgumentParser(description='Atualizar o bot EVA & GUARANI')
    parser.add_argument('--check-only', action='store_true', help='Apenas verificar se há atualizações, sem aplicá-las')
    parser.add_argument('--force', action='store_true', help='Forçar atualização mesmo se não houver mudanças detectadas')
    parser.add_argument('--no-restart', action='store_true', help='Não reiniciar o bot após a atualização')
    parser.add_argument('--no-backup', action='store_true', help='Não fazer backup antes da atualização')
    args = parser.parse_args()

    print("=" * 50)
    print("EVA & GUARANI - Atualizador Automático")
    print("=" * 50)

    # Verificar se é um repositório Git
    if not is_git_repository():
        print("❌ Este diretório não é um repositório Git.")
        print("A atualização automática só funciona com instalações baseadas em Git.")
        return 1

    # Verificar se há atualizações
    print("\nVerificando atualizações...")
    has_updates, commit_count, updates = check_for_updates()

    if has_updates:
        print(f"✅ {commit_count} atualizações disponíveis:")
        for update in updates:
            print(f"   - {update}")
    else:
        print("✅ O bot já está na versão mais recente.")

        if not args.force:
            if args.check_only:
                return 0
            else:
                print("\nNenhuma atualização necessária.")
                return 0
        else:
            print("\nForçando atualização mesmo sem mudanças detectadas...")

    if args.check_only:
        return 0

    # Fazer backup antes da atualização
    if not args.no_backup:
        print("\nCriando backups antes da atualização...")
        config_backup = backup_config()
        files_backup = backup_bot_files()

        if config_backup and files_backup:
            print("✅ Backups criados com sucesso.")
        else:
            print("⚠️ Houve problemas ao criar os backups.")

            if not args.force:
                print("Atualização cancelada. Use --force para continuar mesmo assim.")
                return 1

    # Atualizar o repositório
    print("\nAtualizando o repositório...")
    success, message = update_repository()

    if success:
        print("✅ Repositório atualizado com sucesso.")

        # Preparar mensagem de notificação
        if has_updates:
            notification_message = f"""
🔄 <b>BOT ATUALIZADO</b> - {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}

<b>Atualizações instaladas:</b> {commit_count}
<b>Detalhes:</b>
"""
            for update in updates[:5]:  # Limitar a 5 atualizações para não sobrecarregar a mensagem
                notification_message += f"- {update}\n"

            if len(updates) > 5:
                notification_message += f"... e mais {len(updates) - 5} atualizações.\n"
        else:
            notification_message = f"""
🔄 <b>BOT ATUALIZADO</b> - {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}

<b>Detalhes:</b> Atualização forçada sem mudanças detectadas.
"""

        # Reiniciar o bot
        if not args.no_restart:
            print("\nReiniciando o bot...")
            if restart_bot():
                print("✅ Bot reiniciado com sucesso.")
                notification_message += "\n<b>Status:</b> Bot reiniciado com sucesso."
            else:
                print("❌ Falha ao reiniciar o bot.")
                notification_message += "\n<b>Status:</b> Falha ao reiniciar o bot. Reinicie manualmente."
        else:
            print("\nReinicialização automática desativada.")
            notification_message += "\n<b>Status:</b> Reinicialização automática desativada. Reinicie manualmente."

        # Enviar notificação
        print("\nEnviando notificação sobre a atualização...")
        if send_notification(notification_message):
            print("✅ Notificação enviada com sucesso.")
        else:
            print("⚠️ Falha ao enviar notificação.")

        print("\n" + "=" * 50)
        print("Atualização concluída com sucesso!")
        if not args.no_restart:
            print("O bot foi reiniciado e está em execução.")
        else:
            print("Lembre-se de reiniciar o bot manualmente: python start_eva_guarani.py --restart")
        print("=" * 50)

        return 0
    else:
        print(f"❌ Falha ao atualizar o repositório: {message}")
        return 1

if __name__ == "__main__":
    sys.exit(main())
```

Vamos criar um script para verificar se há atualizações disponíveis periodicamente e notificar os administradores:

```python:check_updates.py
#!/usr/bin/env python
# -*- coding: utf-8 -*-

"""
EVA & GUARANI - Verificador de Atualizações
Este script verifica se há atualizações disponíveis para o bot e notifica os administradores.
"""

import os
import sys
import json
import logging
import argparse
import subprocess
from datetime import datetime, timedelta

# Configuração de logging
os.makedirs("logs", exist_ok=True)
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler("logs/update_checker.log"),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger("update_checker")

# Carregar configuração
CONFIG_PATH = os.path.join("config", "eva_guarani_config.json")

def load_config():
    """Carrega a configuração do bot."""
    try:
        if not os.path.exists(CONFIG_PATH):
            logger.error(f"Arquivo de configuração não encontrado: {CONFIG_PATH}")
            return None

        with open(CONFIG_PATH, 'r', encoding='utf-8') as f:
            config = json.load(f)

        return config
    except Exception as e:
        logger.error(f"Erro ao carregar configuração: {e}")
        return None

def is_git_repository():
    """Verifica se o diretório atual é um repositório Git."""
    try:
        result = subprocess.run(
            ["git", "rev-parse", "--is-inside-work-tree"],
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
            check=False
        )
        return result.returncode == 0
    except Exception:
        return False

def check_for_updates():
    """Verifica se há atualizações disponíveis no repositório Git."""
    try:
        # Atualizar referências remotas
        subprocess.run(
            ["git", "fetch"],
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            check=True
        )

        # Verificar se há diferenças entre o branch local e o remoto
        result = subprocess.run(
            ["git", "rev-list", "HEAD..origin/main", "--count"],
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
            check=True
        )

        commit_count = int(result.stdout.strip())

        if commit_count > 0:
            # Obter informações sobre as atualizações disponíveis
            log_result = subprocess.run(
                ["git", "log", "HEAD..origin/main", "--pretty=format:%h %s"],
                stdout=subprocess.PIPE,
                stderr=subprocess.PIPE,
                text=True,
                check=True
            )

            updates = log_result.stdout.strip().split('\n')
            return True, commit_count, updates
        else:
            return False, 0, []
    except Exception as e:
        logger.error(f"Erro ao verificar atualizações: {e}")
        return False, 0, []

def get_last_check_time():
    """Obtém a data e hora da última verificação de atualizações."""
    try:
        last_check_file = os.path.join("logs", "last_update_check.txt")

        if not os.path.exists(last_check_file):
            return None

        with open(last_check_file, 'r') as f:
            timestamp = f.read().strip()

        return datetime.fromisoformat(timestamp)
    except Exception as e:
        logger.error(f"Erro ao obter data da última verificação: {e}")
        return None

def save_check_time():
    """Salva a data e hora atual como a última verificação de atualizações."""
    try:
        last_check_file = os.path.join("logs", "last_update_check.txt")

        with open(last_check_file, 'w') as f:
            f.write(datetime.now().isoformat())

        return True
    except Exception as e:
        logger.error(f"Erro ao salvar data da verificação: {e}")
        return False

def send_notification(message):
    """Envia uma notificação sobre atualizações disponíveis."""
    try:
        if os.path.exists("notify_status.py"):
            logger.info("Enviando notificação sobre atualizações disponíveis...")

            result = subprocess.run(
                [sys.executable, "notify_status.py", "--message", message, "--force"],
                stdout=subprocess.PIPE,
                stderr=subprocess.PIPE,
                text=True,
                check=False
            )

            if result.returncode == 0:
                logger.info("Notificação enviada com sucesso.")
                return True
            else:
                logger.error(f"Erro ao enviar notificação: {result.stderr}")
                return False
        else:
            logger.warning("Script de notificação não encontrado.")
            return False
    except Exception as e:
        logger.error(f"Erro ao enviar notificação: {e}")
        return False

def main():
    """Função principal."""
    parser = argparse.ArgumentParser(description='Verificar atualizações para o bot EVA & GUARANI')
    parser.add_argument('--force', action='store_true', help='Forçar verificação mesmo se já foi verificado recentemente')
    parser.add_argument('--interval', type=int, default=24, help='Intervalo mínimo entre verificações em horas (padrão: 24)')
    parser.add_argument('--notify-always', action='store_true', help='Notificar mesmo se não houver atualizações')
    args = parser.parse_args()

    # Verificar se é um repositório Git
    if not is_git_repository():
        logger.error("Este diretório não é um repositório Git.")
        logger.error("A verificação de atualizações só funciona com instalações baseadas em Git.")
        return 1

    # Verificar se já foi verificado recentemente
    if not args.force:
        last_check = get_last_check_time()
        if last_check is not None:
            time_since_last_check = datetime.now() - last_check
            if time_since_last_check < timedelta(hours=args.interval):
                logger.info(f"Última verificação foi há {time_since_last_check.total_seconds() / 3600:.1f} horas.")
                logger.info(f"Aguardando intervalo mínimo de {args.interval} horas.")
                return 0

    # Verificar se há atualizações
    logger.info("Verificando atualizações...")
    has_updates, commit_count, updates = check_for_updates()

    # Salvar hora da verificação
    save_check_time()

    if has_updates:
        logger.info(f"{commit_count} atualizações disponíveis:")
        for update in updates:
            logger.info(f"   - {update}")

        # Preparar mensagem de notificação
        notification_message = f"""
🔄 <b>ATUALIZAÇÕES DISPONÍVEIS</b> - {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}

<b>Atualizações disponíveis:</b> {commit_count}
<b>Detalhes:</b>
"""
        for update in updates[:5]:  # Limitar a 5 atualizações para não sobrecarregar a mensagem
            notification_message += f"- {update}\n"

        if len(updates) > 5:
            notification_message += f"... e mais {len(updates) - 5} atualizações.\n"

        notification_message += """
<b>Para atualizar:</b>
- Windows: Execute update_bot.bat
- Linux/Mac: Execute ./update_bot.sh
- Ou execute: python update_bot.py
"""

        # Enviar notificação
        send_notification(notification_message)
        return 0
    else:
        logger.info("Nenhuma atualização disponível.")

        if args.notify_always:
            notification_message = f"""
✅ <b>BOT ATUALIZADO</b> - {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}

O bot já está na versão mais recente. Nenhuma atualização disponível.
"""
            send_notification(notification_message)

        return 0

if __name__ == "__main__":
    sys.exit(main())
```

```shell script:schedule_checks.sh
#!/bin/bash

# EVA & GUARANI - Agendador de Verificações
# Este script agenda verificações periódicas do bot no Linux usando cron

echo "============================================================"
echo "            EVA & GUARANI - AGENDADOR DE VERIFICAÇÕES"
echo "============================================================"
echo ""
echo "Este script agenda verificações periódicas do bot no Linux usando cron."
echo ""

# Mudar para o diretório do script
cd "$(dirname "$0")"

# Verificar se Python está instalado
if ! command -v python3 &> /dev/null; then
    echo "Python 3 não encontrado! Por favor, instale o Python 3.8 ou superior."
    echo "No Ubuntu/Debian: sudo apt install python3 python3-pip"
    echo "No Fedora: sudo dnf install python3 python3-pip"
    exit 1
fi

# Verificar se os scripts existem
if [ ! -f "check_bot_status.py" ]; then
    echo "Script check_bot_status.py não encontrado!"
    exit 1
fi

if [ ! -f "notify_status.py" ]; then
    echo "Script notify_status.py não encontrado!"
    exit 1
fi

# Verificar se o script de atualizações existe
if [ ! -f "check_updates.py" ]; then
    echo "Script check_updates.py não encontrado!"
    echo "As verificações de atualizações não serão agendadas."
    NO_UPDATE_CHECK=1
else
    NO_UPDATE_CHECK=0
fi

# Obter caminho absoluto dos scripts
SCRIPT_DIR="$(pwd)"
CHECK_SCRIPT="${SCRIPT_DIR}/check_bot_status.py"
NOTIFY_SCRIPT="${SCRIPT_DIR}/notify_status.py"
UPDATE_SCRIPT="${SCRIPT_DIR}/check_updates.py"

# Verificar se o usuário tem permissão para editar o crontab
if ! crontab -l &> /dev/null && [ "$?" -ne 0 ] && [ "$?" -ne 1 ]; then
    echo "Você não tem permissão para editar o crontab."
    echo "Execute este script com privilégios adequados."
    exit 1
fi

# Menu de opções
echo "Escolha o intervalo de verificação:"
echo "1. A cada hora"
echo "2. A cada 3 horas"
echo "3. A cada 6 horas"
echo "4. A cada 12 horas"
echo "5. Uma vez por dia"
echo "6. Remover agendamentos existentes"
echo ""

read -p "Digite o número da opção desejada: " choice

# Remover tarefas existentes
CURRENT_CRONTAB=$(crontab -l 2>/dev/null | grep -v "check_bot_status.py\|notify_status.py\|check_updates.py")

if [ "$choice" == "6" ]; then
    echo "$CURRENT_CRONTAB" | crontab -
    echo "Agendamentos removidos com sucesso."
    exit 0
fi

# Definir expressão cron com base na escolha
case $choice in
    1)
        # A cada hora
        CHECK_SCHEDULE="0 * * * *"
        NOTIFY_SCHEDULE="30 * * * *"
        UPDATE_SCHEDULE="0 */6 * * *"  # A cada 6 horas
        ;;
    2)
        # A cada 3 horas
        CHECK_SCHEDULE="0 */3 * * *"
        NOTIFY_SCHEDULE="30 */3 * * *"
        UPDATE_SCHEDULE="0 */12 * * *"  # A cada 12 horas
        ;;
    3)
        # A cada 6 horas
        CHECK_SCHEDULE="0 */6 * * *"
        NOTIFY_SCHEDULE="30 */6 * * *"
        UPDATE_SCHEDULE="0 8 * * *"  # Uma vez por dia às 8h
        ;;
    4)
        # A cada 12 horas
        CHECK_SCHEDULE="0 */12 * * *"
        NOTIFY_SCHEDULE="30 */12 * * *"
        UPDATE_SCHEDULE="0 8 * * *"  # Uma vez por dia às 8h
        ;;
    5)
        # Uma vez por dia
        CHECK_SCHEDULE="0 8 * * *"
        NOTIFY_SCHEDULE="0 20 * * *"
        UPDATE_SCHEDULE="0 12 * * *"  # Uma vez por dia às 12h
        ;;
    *)
        echo "Opção inválida!"
        exit 1
        ;;
esac

# Adicionar novas tarefas ao crontab
NEW_CRONTAB="${CURRENT_CRONTAB}"
NEW_CRONTAB="${NEW_CRONTAB}
# EVA & GUARANI - Verificações automáticas
${CHECK_SCHEDULE} cd ${SCRIPT_DIR} && python3 ${CHECK_SCRIPT} > ${SCRIPT_DIR}/logs/cron_check.log 2>&1
${NOTIFY_SCHEDULE} cd ${SCRIPT_DIR} && python3 ${NOTIFY_SCRIPT} > ${SCRIPT_DIR}/logs/cron_notify.log 2>&1
"

# Adicionar verificação de atualizações se o script existir
if [ $NO_UPDATE_CHECK -eq 0 ]; then
    NEW_CRONTAB="${NEW_CRONTAB}
# EVA & GUARANI - Verificação de atualizações
${UPDATE_SCHEDULE} cd ${SCRIPT_DIR} && python3 ${UPDATE_SCRIPT} > ${SCRIPT_DIR}/logs/cron_updates.log 2>&1
"
fi

# Atualizar crontab
echo "$NEW_CRONTAB" | crontab -

echo ""
echo "Verificações agendadas com sucesso!"
if [ $NO_UPDATE_CHECK -eq 0 ]; then
    echo "Verificações de status, notificações e atualizações foram agendadas."
else
    echo "Verificações de status e notificações foram agendadas."
    echo "Verificações de atualizações não foram agendadas (script não encontrado)."
fi
echo "Para visualizar as tarefas agendadas, execute: crontab -l"
echo ""
```

- O agendamento permite automatizar todo o processo, garantindo verificações periódicas

Os administradores agora têm controle total sobre o processo de atualização, podendo escolher quando e como atualizar o bot, com opções para:
- Apenas verificar atualizações sem aplicá-las
- Atualizar com ou sem backup
- Atualizar com ou sem reinicialização automática
- Forçar atualizações mesmo sem mudanças detectadas

Isso torna a manutenção do bot muito mais simples e segura, reduzindo o risco de problemas durante atualizações.
